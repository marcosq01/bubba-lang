func int partition(int : low, int : high) {
    vars {
        int : pivot, i, j, temp;
    }

    pivot = A[high];
    i = low - 1;
    j = low;
    while (j < high) {
        if (A[j] <= pivot) {
            i = i + 1;

            temp = A[i];
            A[i] = A[j];
            A[j] = temp;

        };

        j = j + 1;
    }

    temp = A[i + 1];
    A[i + 1] = A[high];
    A[high] = temp;

    return i + 1;

}

func void qs(int : low, int : high) {
    vars {
        int : pi;
    }
    if (low < high) {
        pi = partition(low, high);
        qs(low, pi - 1);
        qs(pi + 1, high);
    };
}
