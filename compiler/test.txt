prog Factorial:

vars {
    int : recursive, iterative, n, x, y, z, a;
}

func int factorialRecursion(int : n) {
    if (n <= 1) {
        return 1;
    };
    return n * factorialRecursion(n - 1);
}


func int factorialIterative(int : n) {
    vars {
        int : i, res;
    }

    i = 1;
    res = 1;

    while (i <= n) {
        res = res * i;
        i = i + 1;
    }

    return res;
}

main() {
    y = 1;
    n = 4;
    a = 2;
    for x in 1..n+1 step y {
        for z in 1..6 step a {
            print(z);
        }
        print("-----");
    }
}
